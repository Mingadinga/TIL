```sql
# 프로그래머스 문제 풀이 기록
# 1. 평균 일일 대여 요금 구하기
select round(avg(DAILY_FEE), 0) as AVERAGE_FEE
from CAR_RENTAL_COMPANY_CAR
where CAR_TYPE = 'SUV'

# 2. 인기 있는 아이스크림
SELECT FLAVOR
FROM FIRST_HALF
ORDER BY
    TOTAL_ORDER DESC,
    SHIPMENT_ID ASC;

# 3. 조건에 맞는 도서 리스트 출력하기
SELECT BOOK_ID, TO_CHAR(PUBLISHED_DATE, 'YYYY-MM-DD') as PUBLISHED_DATE
FROM BOOK
WHERE
    TO_CHAR(PUBLISHED_DATE, 'yyyy') = 2021
    AND
    CATEGORY = '인문'
ORDER BY PUBLISHED_DATE ASC;

# 4. 12세 이하인 여자 환자 목록 출력하기
SELECT PT_NAME, PT_NO, GEND_CD, AGE, NVL(TLNO, 'NONE') as TLNO
FROM PATIENT
WHERE AGE <= 12 AND GEND_CD = 'W'
ORDER BY AGE DESC, PT_NAME ASC;

# 5. 과일로 만든 아이스크림 고르기
SELECT A.FLAVOR
FROM FIRST_HALF A, ICECREAM_INFO B
WHERE
    A.FLAVOR = B.FLAVOR AND
    A.TOTAL_ORDER > 3000 AND
    B.INGREDIENT_TYPE = 'fruit_based';

# 6. 흉부외과 또는 일반외과 의사 목록 출력하기
SELECT DR_NAME, DR_ID, MCDP_CD, TO_CHAR(HIRE_YMD, 'YYYY-MM-DD') AS HIRE_YMD
FROM DOCTOR
WHERE
    MCDP_CD = 'CS' OR
    MCDP_CD = 'GS'
ORDER BY
    HIRE_YMD DESC,
    DR_NAME ASC;

# 7. 3월에 태어난 여성 회원 목록 출력하기
SELECT MEMBER_ID, MEMBER_NAME, GENDER, TO_CHAR(DATE_OF_BIRTH, 'YYYY-MM-DD') as DATE_OF_BIRTH
FROM MEMBER_PROFILE
WHERE
    TO_CHAR(DATE_OF_BIRTH, 'MM') = 3 AND
    GENDER = 'W' AND
    TLNO IS NOT NULL
ORDER BY MEMBER_ID ASC;

# 8. 강원도에 위치한 생산공장 목록 출력하기
SELECT FACTORY_ID, FACTORY_NAME, ADDRESS
FROM FOOD_FACTORY
WHERE ADDRESS LIKE '%강원도%'
ORDER BY FACTORY_ID ASC;

# 9. 서울에 위치한 식당 목록 출력하기
SELECT
    I.REST_ID,
    I.REST_NAME,
    I.FOOD_TYPE,
    I.FAVORITES,
    I.ADDRESS,
    ROUND(AVG(R.REVIEW_SCORE),2) as SCORE
FROM REST_INFO I, REST_REVIEW R
WHERE
    I.REST_ID = R.REST_ID AND
    I.ADDRESS LIKE '서울%'
GROUP BY I.REST_ID, I.REST_NAME, I.FOOD_TYPE, I.FAVORITES, I.ADDRESS
ORDER BY SCORE DESC, I.FAVORITES DESC

```
